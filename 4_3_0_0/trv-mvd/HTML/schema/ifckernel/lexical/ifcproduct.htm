EXPRESS Specification
ENTITY IfcProduct
 ABSTRACT SUPERTYPE OF(ONEOF(IfcElement, IfcSpatialElement))
 SUBTYPE OF (IfcObject);
  ObjectPlacement : OPTIONAL IfcStrippedOptional;
  Representation : OPTIONAL IfcStrippedOptional;
 INVERSE
 WHERE
  PlacementForShapeRepresentation : (EXISTS(Representation) AND EXISTS(ObjectPlacement))
OR (EXISTS(Representation) AND
    (SIZEOF(QUERY(temp <* Representation.Representations | 'IFCREPRESENTATIONRESOURCE.IfcStrippedOptional' IN TYPEOF(temp))) = 0))
OR (NOT(EXISTS(Representation)));
END_ENTITY;
